const video = document.getElementById('video'); const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); async function setupCamera() { try { const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: 'environment' }, audio: false }); video.srcObject = stream; return new Promise((resolve) => { video.onloadedmetadata = () => { video.play(); resolve(video); }; }); } catch (e) { alert("هەڵە لە کامێرا: " + e.message); } } async function startDetection() { // نیشاندانی نامەیەک تا مۆدێلەکە بار دەبێت console.log("Loading AI Model..."); const net = await bodyPix.load({ architecture: 'MobileNetV1', outputStride: 16, multiplier: 0.50, quantBytes: 2 }); canvas.width = video.videoWidth; canvas.height = video.videoHeight; async function detect() { const segmentation = await net.segmentPerson(video); const maskColor = { r: 0, g: 255, b: 255, a: 150 }; const background = { r: 0, g: 0, b: 0, a: 0 }; const coloredPartImage = bodyPix.toMask(segmentation, maskColor, background); ctx.clearRect(0, 0, canvas.width, canvas.height); bodyPix.drawMask(canvas, video, coloredPartImage, 0.7, 0, false); requestAnimationFrame(detect); } detect(); } setupCamera().then(startDetection);
